// Fixed article reading functionality function readArticle(articleId) { console.log('Reading article:', articleId); // Hide the article listing const listing = document.getElementById('articles-listing'); const articleView = document.getElementById('article-full-view'); if (listing && articleView) { listing.style.display = 'none'; articleView.style.display = 'block'; // Load article content const article = articlesDatabase[articleId]; if (article) { articleView.innerHTML = article.content; window.scrollTo({ top: 0, behavior: 'smooth' }); } } else { console.error('Article elements not found!'); } } function backToArticles() { const listing = document.getElementById('articles-listing'); const articleView = document.getElementById('article-full-view'); if (listing && articleView) { listing.style.display = 'block'; articleView.style.display = 'none'; window.scrollTo({ top: 0, behavior: 'smooth' }); } } // Update your article buttons to use onclick instead of href document.querySelectorAll('.read-article-btn').forEach(btn => { btn.addEventListener('click', function(e) { e.preventDefault(); // Prevent default link behavior const articleId = this.getAttribute('data-article'); readArticle(articleId); }); });
